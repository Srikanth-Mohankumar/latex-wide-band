% Baseline for comparison with test-wideband.tex.
% Shows the two classic approaches to full-width material in twocolumn mode:
%   page 1 -- cuted's \begin{strip} (closest existing tool, TeX-only OR hack)
%   page 2 -- the \onecolumn/\twocolumn switch (what naive widetext-style
%             emulation does: flushes the page and restarts columns)
\documentclass[twocolumn]{article}
\usepackage{amsmath}
\usepackage{lipsum}
\usepackage{cuted}
\setlength{\columnseprule}{0.2pt}

\begin{document}

\section{cuted strip}
\lipsum[1][1-8]

\begin{strip}
\begin{equation}
  \mathcal{Z}[J] = \int \mathcal{D}\phi \,
  \exp\!\Bigl( i\!\int\! d^4x \,\bigl[\tfrac12 \partial_\mu\phi\,\partial^\mu\phi
  - \tfrac12 m^2\phi^2 - \tfrac{\lambda}{4!}\phi^4 + J\phi\bigr] \Bigr)
  \qquad\text{(cuted strip)}
\end{equation}
\end{strip}

\lipsum[2-4]
\lipsum[5-7]

\clearpage
\section{Column switch (widetext-style emulation)}
\lipsum[1][1-8]

\onecolumn

  - \tfrac12 m^2\phi^2 - \tfrac{\lambda}{4!}\phi^4 + J\phi\bigr] \Bigr)
  \qquad\text{(\string\onecolumn\ switch: page is flushed)}
\end{equation}
\twocolumn

\lipsum[2-4]

\end{document}
